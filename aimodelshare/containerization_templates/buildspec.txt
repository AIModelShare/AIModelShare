version: 0.2

phases:
  pre_build:
    commands:
      - aws configure set aws_access_key_id $aws_access_key_id
      - aws configure set aws_secret_access_key $aws_secret_access_key
      - aws ecr get-login-password --region $region | docker login --username AWS --password-stdin $account_id.dkr.ecr.$region.amazonaws.com
  build:
    commands:
      - docker build --label $label --tag $repository:$image_tag .
      - docker tag aimoodelshare-base-image-public:latest $account_id.dkr.ecr.$region.amazonaws.com/$repository:$image_tag
  post_build:
    commands:
      - docker push $account_id.dkr.ecr.$region.amazonaws.com/$repository:$image_tag
